31  public static void print_class (Class<?> c) {
32  // Print modifiers, type (class or interface), name and superclass.
33  if (c.isInterface()) {
34  // The modifiers will include the "interface" keyword here...
35  System.out.print(Modifier.toString(c.getModifiers()) + " " + c.getName());
36  }
37  else {
38  System.out.print(Modifier.toString(c.getModifiers()) + " class " + c.getName());
39  if (c.getSuperclass() != null) {
40  System.out.print(" extends " + c.getSuperclass().getName());
41  }
42  }
43 
44  // Print interfaces or super-interfaces of the class or interface.
45  Class<?>[] interfaces = c.getInterfaces();
46  if ((interfaces != null) && (interfaces.length > 0)) {
47  if (c.isInterface()) {
48  System.out.println(" extends ");
49  }
50  else {
51  System.out.print(" implements ");
52  }
53  for (int i = 0; i < interfaces.length; i++) {
54  if (i > 0) {
55  System.out.print(", ");
56  }
57  System.out.print(interfaces[i].getName());
58  }
59  }

118  public static void print_method_or_constructor (Member member) {
119  Class<?> returntype = null, parameters[], exceptions[];
120  if (member instanceof Method) {
121  Method m = (Method) member;
122  returntype = m.getReturnType();
123  parameters = m.getParameterTypes();
124  exceptions = m.getExceptionTypes();
125  }
126  else {
127  Constructor<?> c = (Constructor<?>) member;
128  parameters = c.getParameterTypes();
129  exceptions = c.getExceptionTypes();
130  }
131 
132  System.out.print(" " + modifiers(member.getModifiers()) +
133  ((returntype != null) ? typename(returntype) + " " : "") +
134  member.getName() + "(");
135  for (int i = 0; i < parameters.length; i++) {
136  if (i > 0) {
137  System.out.print(", ");
138  }
139  System.out.print(typename(parameters[i]));
140  }
141  System.out.print(")");
142  if (exceptions.length > 0) {
143  System.out.print(" throws ");
144  }
145  for (int i = 0; i < exceptions.length; i++) {
146  if (i > 0) {
